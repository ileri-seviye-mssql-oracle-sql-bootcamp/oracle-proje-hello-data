-- movie
CREATE TABLE cinema.movie
  (
    movie_id NUMBER GENERATED BY DEFAULT AS IDENTITY START WITH 100 PRIMARY KEY,
    movie_title VARCHAR2(255) NOT NULL,
    movie_director VARCHAR2(255) NOT NULL,
    movie_cast VARCHAR2(255) NOT NULL,
    movie_duration  NUMBER NOT NULL,
    movie_description VARCHAR2(2000) NOT NULL
  );
  
-- screening
CREATE TABLE cinema.screening
  (
    screening_id NUMBER GENERATED BY DEFAULT AS IDENTITY START WITH 100 PRIMARY KEY,
    movie_id NUMBER NOT NULL,  -- fk
    auditorium_id NUMBER NOT NULL, --fk
    screening_star TIMESTAMP, 
    CONSTRAINT fk_screening_movie FOREIGN KEY( movie_id )
      REFERENCES cinema.movie( movie_id )
      ON DELETE CASCADE,
    CONSTRAINT fk_screening_auditorium FOREIGN KEY( auditorium_id )
      REFERENCES cinema.auditorium ( auditorium_id )
      ON DELETE CASCADE
  );
  
  
-- auditorium
CREATE TABLE cinema.auditorium
  (
    auditorium_id NUMBER GENERATED BY DEFAULT AS IDENTITY START WITH 100 PRIMARY KEY,
    auditorium_name VARCHAR2(255) NOT NULL,
    seats_no NUMBER NOT NULL
  );
  
  
--seat
CREATE TABLE cinema.seat
  (
    seat_id NUMBER GENERATED BY DEFAULT AS IDENTITY START WITH 100 PRIMARY KEY,
    seat_row VARCHAR2(255) NOT NULL,
    seat_number NUMBER NOT NULL,
    auditorium_id NUMBER NOT NULL,  -- fk
    CONSTRAINT fk_seat_auditorium FOREIGN KEY( auditorium_id )
      REFERENCES cinema.auditorium( auditorium_id )
      ON DELETE CASCADE
  );
  
--reservation type
CREATE TABLE cinema.reservation_type
  (
    reservation_type_id NUMBER GENERATED BY DEFAULT AS IDENTITY START WITH 100 PRIMARY KEY,
    reservation_type VARCHAR2(255) NOT NULL
  );
  
--employee
CREATE TABLE cinema.employee
  (
    employee_id NUMBER GENERATED BY DEFAULT AS IDENTITY START WITH 100 PRIMARY KEY,
    username VARCHAR2(255) NOT NULL,
    password VARCHAR2(255) NOT NULL
  );
  
--reservation
CREATE TABLE cinema.reservation
(
    reservation_id NUMBER GENERATED BY DEFAULT AS IDENTITY START WITH 100 PRIMARY KEY,
    screening_id NUMBER NOT NULL,  -- fk
    employee_reserved_id NUMBER NOT NULL,  -- fk
    reservation_type_id NUMBER NOT NULL,  -- fk
    reservation_contact VARCHAR2(255) NOT NULL,
    reserved VARCHAR2(5) NOT NULL,
    employee_paid_id NUMBER NOT NULL,  -- fk
    paid VARCHAR2(5) NOT NULL,
    active VARCHAR2(5) NOT NULL,
    CONSTRAINT fk_reservation_screening FOREIGN KEY( screening_id )
      REFERENCES cinema.screening ( screening_id )
      ON DELETE CASCADE,
    CONSTRAINT fk_reservation_employee FOREIGN KEY( employee_reserved_id )
      REFERENCES cinema.employee( employee_id )
      ON DELETE CASCADE,
    CONSTRAINT fk_reservation_reservation_type FOREIGN KEY( reservation_type_id )
      REFERENCES cinema.reservation_type( reservation_type_id )
      ON DELETE CASCADE,
    CONSTRAINT fk_reservationemployee FOREIGN KEY( employee_paid_id )
      REFERENCES cinema.employee( employee_id )
      ON DELETE CASCADE
); 
  
--seat reserved
CREATE TABLE cinema.seat_reserved
  (
    seat_reserved_id NUMBER GENERATED BY DEFAULT AS IDENTITY START WITH 100 PRIMARY KEY,
    seat_id NUMBER NOT NULL,  -- fk
    reservation_id NUMBER NOT NULL,  -- fk
    screening_id NUMBER NOT NULL,  -- fk
    CONSTRAINT fk_seat_reserved_seat FOREIGN KEY( seat_id )
      REFERENCES cinema.seat( seat_id )
      ON DELETE CASCADE,
    CONSTRAINT fk_seat_reserved_reservation FOREIGN KEY( reservation_id )
      REFERENCES cinema.reservation( reservation_id )
      ON DELETE CASCADE,
    CONSTRAINT fk_seat_reserved_screening FOREIGN KEY( screening_id )
      REFERENCES cinema.screening ( screening_id )
      ON DELETE CASCADE
  );
